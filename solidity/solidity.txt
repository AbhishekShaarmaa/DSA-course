//           -----/ STRUCTURE /-----

 struct student{
     uint roll ; 
     string name ;
 }

contract demo 
{
   student public s1;

   constructor (uint _rollno , string memory _name)
   {
        s1.name = _name;
        s1.roll = _rollno;
   }

   function change( uint _rollno , string memory _name )   public 
   {
       student memory newstudent =  student ( {
           roll : _rollno , 
           name  : _name 
       }); 
   }
}
